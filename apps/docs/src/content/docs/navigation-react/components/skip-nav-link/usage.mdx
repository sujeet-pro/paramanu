The Skip Nav Link is an accessibility component that lets keyboard users bypass repetitive navigation to reach main content.

## Basic Usage

<ComponentPreview
  htmlCode={`<a href="#main-content" class="pm-skip-link">Skip to main content</a>
<nav><!-- navigation --></nav>
<main id="main-content">Main content</main>`}
  reactCode={`import { SkipLink } from "@paramanu/navigation-react"

export default function App() {
  return (
    <>
      <SkipLink />
      <nav>{/* navigation */}</nav>
      <main id="main-content">Main content</main>
    </>
  )
}`}
/>
