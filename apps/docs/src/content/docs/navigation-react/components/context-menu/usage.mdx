The Context Menu appears on right-click and provides contextual actions for the target element.

## Basic Usage

<ComponentPreview
  htmlCode={`<div class="pm-ctx-menu pm-ctx-menu--md pm-ctx-menu--open" role="menu">
  <div role="menuitem">Cut</div>
  <div role="menuitem">Copy</div>
  <div role="menuitem">Paste</div>
</div>`}
  reactCode={`import { CtxMenu } from "@paramanu/navigation-react"

export default function App() {
  return (
    <CtxMenu open>
      <div role="menuitem">Cut</div>
      <div role="menuitem">Copy</div>
      <div role="menuitem">Paste</div>
    </CtxMenu>
  )
}`}
/>
