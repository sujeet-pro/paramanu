List renders ordered or unordered lists with configurable marker style, spacing, and an unstyled option.

## Basic Usage

<ComponentPreview
  htmlCode={`<ul class="pm-list pm-list--unordered pm-list--spacing-md">
  <li>First item</li>
  <li>Second item</li>
  <li>Third item</li>
</ul>`}
  reactCode={`import { List } from "@paramanu/typography-react"

export default function App() {
  return (
    <List>
      <li>First item</li>
      <li>Second item</li>
      <li>Third item</li>
    </List>
  )
}`}
/>
