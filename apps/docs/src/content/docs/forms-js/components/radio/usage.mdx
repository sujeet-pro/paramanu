The Radio component provides single-selection radio buttons, typically used within a RadioGroup.

## Basic Usage

<ComponentPreview
  htmlCode={`<div class="pm-radio-group pm-radio-group--vertical pm-radio-group--md" role="radiogroup">\n  <label class="pm-radio pm-radio--md">\n    <input type="radio" class="pm-radio__input" name="fruit" value="apple" />\n    <span class="pm-radio__indicator"></span>\n    <span class="pm-radio__label">Apple</span>\n  </label>\n  <label class="pm-radio pm-radio--md">\n    <input type="radio" class="pm-radio__input" name="fruit" value="banana" />\n    <span class="pm-radio__indicator"></span>\n    <span class="pm-radio__label">Banana</span>\n  </label>\n</div>`}
  reactCode={`import { Radio, RadioGroup } from "@paramanu/forms-react"\n\nexport default function App() {\n  return (\n    <RadioGroup name="fruit">\n      <Radio name="fruit" value="apple">Apple</Radio>\n      <Radio name="fruit" value="banana">Banana</Radio>\n    </RadioGroup>\n  )\n}`}
/>
