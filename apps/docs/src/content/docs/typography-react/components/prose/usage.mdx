Prose applies consistent typographic styling to nested HTML content such as headings, paragraphs, lists, and code blocks.

## Basic Usage

<ComponentPreview
  htmlCode={`<div class="pm-prose pm-prose--md">
  <h2>Article Title</h2>
  <p>This content is styled by the prose container.</p>
  <ul><li>First item</li><li>Second item</li></ul>
</div>`}
  reactCode={`import { Prose } from "@paramanu/typography-react"

export default function App() {
  return (
    <Prose>
      <h2>Article Title</h2>
      <p>This content is styled by the prose container.</p>
      <ul><li>First item</li><li>Second item</li></ul>
    </Prose>
  )
}`}
/>
