## Basic Usage

<ComponentPreview
  htmlCode={`<div class="pm-center" style="height: 200px; border: 1px dashed #ccc;">
  <div style="background: #e2e8f0; padding: 16px;">Centered</div>
</div>`}
  reactCode={`import { Center } from "@paramanu/primitives-react"

export default function App() {
  return (
    <Center style={{ height: 200, border: "1px dashed #ccc" }}>
      <div>Centered</div>
    </Center>
  )
}`}
/>
